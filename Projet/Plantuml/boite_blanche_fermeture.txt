@startuml
actor Utilisateur as U 
participant "Agent Local:Agent" as A 
participant "X:Agent" as X 

alt Utilisateur ferme session 
U --> A : fermer session
A --> X : fermeture 
X --> A : fermeture OK 
A --> U : session fermée avec succès 
else X ferme session 
X --> A : fermeture 
A --> X : fermeture OK 
A --> U : session fermée par X 
end 
@enduml
